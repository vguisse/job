\begin{exercice}
 Soit $(u_n)$ la\baremeexo{3} suite définie par $u_1=1$, et pour tout $n\in\N^*$, $u_{n+1}=u_n+n+1$.
Démontrer par récurrence que pour tout $n\in\N^*$, $u_n=\dfrac{n(n+1)}{2}$. 
\end{exercice}
\vfill
\begin{exercice}
 Soit $(u_n)$ la\baremeexo{3} suite définie par $u_0=9$, et pour tout $n\in\N^*$, $u_{n+1}=1+\sqrt{u_n}$.
Démontrer par récurrence que $(u_n)$ est décroissante.
\end{exercice}
