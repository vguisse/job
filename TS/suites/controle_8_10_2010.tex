\begin{exercice}
Soit\baremeexo{4} $(u_n)$ la suite définie par $u_0=1$ et, pour
tout entier $n\geq0$, $u_{n+1}=5u_n-8$.
\begin{enumerate}
 \item Calculer\baremeque{0,5} $u_1$, $u_2$ et $u_3$.
 \item Soit $(v_n)$ la suite définie par $v_n=u_n-2$ pour $n\in\N$. Prouver que $(v_n)$
est géométrique, et\baremeque{1,5} préciser sa raison.
 \item En déduire l'expression\baremeque{2} de $v_n$, puis de celle $u_n$ en fonction de $n$ pour $n\in\N$.
\end{enumerate}
\end{exercice}

\begin{exercice}[Cours]
 On rapelle\baremeexo{3} que dire qu'une suite tend vers $+\infty$ signifie :

\emph{Pour tout nombre réel $A$, il existe un rang à partir duquel 
tous les termes de la suite sont supérieurs
à $A$.}

\begin{enumerate}
 \item Démontrer\baremeque{2} le théorème suivant : si une suite est croissante et non majorée, alors elle tend vers $+\infty$.
 \item Donner un exemple\baremeque{0,5} de suite non majorée qui ne tend pas vers $+\infty$.
 \item Donner un exemple\baremeque{0,5} de suite croissante qui ne tend pas vers $+\infty$.
\end{enumerate}


\end{exercice}
