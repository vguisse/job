\begin{exercice}
 Soit\baremeexo{6} $(d)$ la droite d'équation $y=2x+1$ et $A(0,1)$ et $B(8,13)$ deux points.
\begin{enumerate}
 \item Faire\baremeque{1} une figure (on donnera les calculs permettant de tracer $(d)$.
 \item Déterminer par le calcul une\baremeque{2} équation de la droite $(AB)$.
 \item Les droites\baremeque{1} $(d)$ et $(AB)$ sont elles parallèles? Justifier.
 \item Déterminer\baremeque{2} les coordonnées du point d'intersection des droites $(d)$ et $(AB)$.
\end{enumerate}

\end{exercice}
