\begin{exercice}
 Soit $f$\baremeexo{4} la fonction définie sur $\R$ par $f(x)=7x-5$.
\begin{enumerate}
 \item Calculer\baremeque{2} $f(10)$, $f(\dfrac{3}{10})$ et $f(\sqrt{11})$.
 \item Donner\baremeque{2} le (ou les) antécédent(s) de 44 par $f$.
\end{enumerate}
\end{exercice}

\begin{exercice}
 Soit $g$\baremeexo{4} la fonction définie sur $\R$ par $g(x)=(x-5)^2-2$.
\begin{enumerate}
 \item Calculer\baremeque{2} $g(2)$ et $g(-3)$.
 \item Quelle\baremeque{1} est l'image de 5 par $g$ ?
 \item Donner\baremeque{1} un antécédent de $-2$ par $g$.
\end{enumerate}
\end{exercice}
