\begin{exercice}[Cours]
Soit $f$ une fonction définie sur un intervalle $I$.
Quelle est la signification de la phrase : \og la fonction $f$ est croissante
sur l'intervalle $I$\fg{} ?
\vfill
\end{exercice}



\begin{exercice}
Soit\baremeexo{4} $f$ une fonction définie sur l'intervalle $[-2;5]$. On sait que
$f$ est strictement décroissante sur l'intervalle $[-2;1]$,
strictement croissante sur l'intervalle $[1;2]$, et strictement
décroissante sur l'intervalle $[2;5]$.
\begin{enumerate}
  \item Comparer\baremeque{1} $f(-2)$ et $f(-1)$, en justifiant soigneusement
  la réponse.
  \vfill
  \item Donner\baremeque{2} le tableau de variations de $f$.
  \vfill
  \item On\baremeque{1} admet que $f(0)=f(2)$. Démontrer que $f(-2)>f(5)$.
  \vfill
\end{enumerate}
\end{exercice}
