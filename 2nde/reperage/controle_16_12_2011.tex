\begin{exercice}[Cours] \baremeexo{8}
\begin{enumerate}
      \item Connaître les formules du cours. Dans le plan muni d'un repère $\Oij$, on note $A(x_A;y_A)$
      et $B(x_B;y_B)$ les coordonnées des points $A$ et $B$. 
      \begin{enumerate}
      \item Quelle\baremeque{1} formule donne les coordonnées du vecteur $\vect{AB}$? 
      \item Quelle\baremeque{1} formule donne les coordonnées du milieu $M$ du segment $[AB]$ ?
      \item Recopier\baremeque{1} et compléter cette équivalence logique vue en cours : \og Deux vecteurs sont colinéaires si et seulement si \dots\fg
      \end{enumerate}
      \item Applications directes. Dans le plan muni d'un repère $\Oij$, on donne $E(-9;6)$, 
	    $F(0;4)$, $G(5;0)$ et $H(3;-4)$.
	  \begin{enumerate}
	    \item Faire\baremeque{1} une figure.
	    \item Calculer\baremeque{1} les coordonnées du milieu $I$ de $[EH]$.
	    \item Calculer\baremeque{1,5} les coordonnées du vecteur $\vect{EH}$, et donner sans justification les coordonnées du vecteur $\vect{FG}$.
	    \item Les vecteurs\baremeque{1} $\vect{EH}$ et $\vect{FG}$ sont-ils colinéaires ? Justifier.
	    \item Que peut-on\baremeque{0,5} en déduire pour les droites $(EH)$ et $(FG)$?
	  \end{enumerate}
\end{enumerate}

\end{exercice}

\begin{exercice}
 Le plan\baremeexo{5} est muni d'un repère orthonormé $\Oij$.
\begin{enumerate}
 \item Sur\baremeque{1} une première figure :
  \begin{enumerate}
    \item Hachurer en bleu l'ensemble $E$ des points $M(x,y)$
tels que $x\geq1$.
    \item Hachurer en rouge l'ensemble $F$ des points $M(x,y)$
tels que $y\geq1$.
   \end{enumerate}
  \item Sur\baremeque{1} une deuxième figure :
\begin{enumerate}
 \item Hachurer en vert l'ensemble $G$ des points $M(x,y)$
tels que $x\geq1$ ET $y\geq1$.
  \item $H$ est l'ensemble des points $M(x,y)$ de $G$ tels que
$x$ et $y$ sont des nombres entiers. Marquer en noir les points de $H$ visibles sur votre figure.
\end{enumerate}
  \item Vrai ou faux\baremeque{3} (justifier la réponse) : Si un cercle de centre $O$ a pour 
rayon un nombre entier, alors il ne passe par aucun point de $H$. 
\end{enumerate}

\end{exercice}
