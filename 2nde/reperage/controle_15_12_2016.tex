\begin{exercice}[Cours]
Dans\baremeexo{4} le plan muni d'un repère $\Oij$,
on donne $A(x_A;y_A)$ et $B(x_B;y_B)$.
\begin{enumerate}
  \item Exprimer les coordonnées du milieu $M$
  du segment $[AB]$ en fonction des coordonnées de 
  $A$ et de $B$.\baremeque{1}
  \item Exprimer les coordonnées du vecteur $\vect{AB}$
  en fonction des coordonnée  de $A$ et de $B$.\baremeque{1}
  \item À quelle \baremeque{2} condition peut-on exprimer la longueur 
  $AB$ en fonction des coordonnées de $A$ et de $B$ ?
  Donner la formule et dire quel théorème permet de la démontrer.
\end{enumerate}
\end{exercice}

 \item Faire\baremeque{1} un figure.

\end{exercice}

\begin{exercice}
 Dans\baremeexo{6} un repère $\Oij$, on considère les points $A(-7;-3)$ et $B(10;2)$.
\begin{enumerate}
 \item Le\baremeque{2} point $C(0;-1)$ appartient-il à la droite $(AB)$? Justifer.
 \item Soit $M(0,y_M)$ le point d'intersection de la droite $(AB)$ avec l'axe des ordonnées (c'est à dire l'axe $(O,\vect{j})$).
  \begin{enumerate}
    \item Exprimer\baremeque{1} les coordonnées de $\vect{AM}$ en fonction de $y_M$.
    \item Montrer\baremeque{1} que $17(y_M+3)=35$ et en déduire la valeur exacte      de $y_M$.
    \end{enumerate}
\item Déterminer\baremeque{2} les coordonnées du point d'intersection de la droite $(AB)$ 
avec l'axe des abscisses.
\end{enumerate}
\end{exercice}
