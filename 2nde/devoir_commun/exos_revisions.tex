\begin{exercice}Connaissance des nombres.
\begin{enumerate}
 \item Rappeler les 5 ensembles de nombres connus en seconde (noms et notations).
 \item \`A quel(s) ensemble(s) de nombres appartiennent les nombres suivants :
\begin{align*}
 12	&&	-\sqrt{49}	&&	\pi	&&	1,5\times10^{5}	&&	1,015	&&	\dfrac{2}{3}	&&	\dfrac{11}{5}	&&	2^{-3}	&&	\sqrt{11}	&&	0,031\times10^{-2}
\end{align*}
 \item Donner si possible l'écriture décimale et l'écriture scientifique des nombres de la question précédente.
\end{enumerate}
\end{exercice}

\begin{exercice}Fonctions affines.
 \begin{enumerate}
  \item Soit $x$ un nombre réel. Peut-on connaître le signe de $2x-3$?
  \item Soit $f$ la fonction définie sur $\R$ par $f(x)=2x-3$.
    \begin{enumerate}
     \item Donner le tableau de variations de $f$ (justifier).
     \item En déduire le tableau de signe de l'expression $2x-3$.
     \item En déduire les solutions de $f(x)<0$.
    \end{enumerate}
 \end{enumerate}
\end{exercice}

\begin{exercice}Fonctions de référence.
  \begin{enumerate}
   \item Donner le tableau de variations et l'allure de la représentation graphique de la fonction inverse.
   \item Donner le tableau de variations et l'allure de la représentation graphique de la fonction carré.
   \item Comparer $-\dfrac{1}{995}$ et $-\dfrac{1}{\numprint{1005}}$ en justifiant.
   \item Comparer $\left(-\dfrac{1}{995}\right)^2$ et $\left(-\dfrac{1}{\numprint{1005}}\right)^2$ (sans calculs) en justifiant.
  \end{enumerate}
\end{exercice}


\begin{exercice}\'Equations de droite.
  \begin{enumerate}
   \item Tracer la droite $(d_1)$ d'équation $y=2x-3$.
   \item Tracer la droite $(d_2)$ passant par $A(2,0)$ et de c\oe fficient directeur $-\dfrac23$.
   \item Tracer la droite $(d_3)$ passant par $B(-1,-3)$ et d'ordonnée à l'origine égale à -1.
   \item Les droites $(d_1)$ et $(d_3)$ sont-elles parallèles? Justifier.
  \end{enumerate}
\end{exercice}

\begin{exercice}Probabilités.\\
 On lance deux dés à 4 faces numérotées 1, 2, 3 et 4. On considère les événements :\\
$D$ : \og on obtient un double\fg\\
$Q$ : \og la somme donne 4\fg
\begin{enumerate}
 \item Déterminer $p(D)$, $p(Q)$ et $p(D\cap Q)$. Calculer $p(\overline{Q})$.
 \item En déduire $p(D\cup Q)$.
\end{enumerate}
\end{exercice}

\begin{exercice}Vecteurs.\\
 Dans un repère $\Oij$, on donne $A(-3;-2)$,  $B(-2;1)$, $C(1;2)$ et $D(0;-1)$.
\begin{enumerate}
 \item Prouver que $ABCD$ est un parallélogramme, puis que c'est un losange.
 \item Calculer les coordonnées du point d'intersection des droites $(AC)$ et $(BD)$.
 \item Le point $M(3,3)$ appartient-t-il à la droite $(BC)$?
\end{enumerate}
\end{exercice}
